rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Users collection - users can read/write their own profile
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Vitamins collection - users can read/write their own vitamin entries
    match /vitamins/{vitaminId} {
      allow read, write: if request.auth != null && resource.data.userId == request.auth.uid;
      allow create: if request.auth != null && request.resource.data.userId == request.auth.uid;
    }
    
    // Food log collection - users can read/write their own food entries
    match /foodLog/{entryId} {
      allow read, write: if request.auth != null && resource.data.userId == request.auth.uid;
      allow create: if request.auth != null && request.resource.data.userId == request.auth.uid;
    }
    
    // User vitamin tablets - users can read/write their own tablets
    match /userTablets/{tabletId} {
      allow read, write: if request.auth != null && resource.data.userId == request.auth.uid;
      allow create: if request.auth != null && request.resource.data.userId == request.auth.uid;
    }
    
    // Daily recommendations - users can read/write their own recommendations
    match /dailyRecommendations/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
  }
}
